  {\Large \textbf{Práctica de Nomenclatura Química (Respuestas)}} \\[4pt]
  \textbf{Óxidos, hidruros, hidrácidos y sales binarias}
\end{center}

\section*{Parte A. Determinar la fórmula a partir del nombre}
\begin{enumerate}
  \item Óxido de hierro(III) $\;\rightarrow\; \mathrm{Fe_2O_3}$
  \item Óxido de cobre(I) $\;\rightarrow\; \mathrm{Cu_2O}$
  \item Hidruro de calcio $\;\rightarrow\; \mathrm{CaH_2}$
  \item Hidruro de estaño(IV) $\;\rightarrow\; \mathrm{SnH_4}$
  \item Hidróxido de aluminio $\;\rightarrow\; \mathrm{Al(OH)_3}$
  \item Hidróxido de plomo(II) $\;\rightarrow\; \mathrm{Pb(OH)_2}$
  \item Ácido clorhídrico $\;\rightarrow\; \mathrm{HCl(ac)}$
  \item Ácido sulfhídrico $\;\rightarrow\; \mathrm{H_2S(ac)}$
  \item Cloruro de sodio $\;\rightarrow\; \mathrm{NaCl}$
  \item Fluoruro de magnesio $\;\rightarrow\; \mathrm{MgF_2}$
\end{enumerate}

\section*{Parte B. Determinar el nombre a partir de la fórmula}
\begin{multicols}{2}
\begin{enumerate}
  \item $\mathrm{FeO}$ $\;\rightarrow\;$ Óxido de hierro(II)
  \item $\mathrm{Fe_2O_3}$ $\;\rightarrow\;$ Óxido de hierro(III)
  \item $\mathrm{NaH}$ $\;\rightarrow\;$ Hidruro de sodio
  \item $\mathrm{BaH_2}$ $\;\rightarrow\;$ Hidruro de bario
  \item $\mathrm{HCl}$ (ac) $\;\rightarrow\;$ Ácido clorhídrico
  \item $\mathrm{H_2S}$ (ac) $\;\rightarrow\;$ Ácido sulfhídrico
  \item $\mathrm{NaCl}$ $\;\rightarrow\;$ Cloruro de sodio
  \item $\mathrm{MgF_2}$ $\;\rightarrow\;$ Fluoruro de magnesio
\end{enumerate}
\end{multicols}

\section*{Parte C. Completar reacciones de formación}
\begin{enumerate}
  \item $\mathrm{2Fe + O_2 \;\rightarrow\; 2FeO}$
  \item $\mathrm{Ca + H_2 \;\rightarrow\; CaH_2}$
  \item $\mathrm{2Na + Cl_2 \;\rightarrow\; 2NaCl}$
  \item $\mathrm{H_2 + S \;\rightarrow\; H_2S}$
\end{enumerate}
